# Task 8: SwiftData Recording Entry Creation - Evidence

## Change Made

Modified: `Scribe/Scribe/Sources/UI/RecordingListView.swift`

### Before:
```swift
private func saveRecording(duration: TimeInterval, fileURL: URL) {
    let filename = fileURL.lastPathComponent
    let id = filename.replacingOccurrences(of: ".m4a", with: "")
    let newRecording = Recording(
        id: id,
        title: "New Recording",
        duration: duration,
        audioFilePath: filename,
        categoryTag: "#NOTE"
    )
```

### After:
```swift
private func saveRecording(duration: TimeInterval, fileURL: URL) {
    let filename = fileURL.lastPathComponent
    let id = filename.replacingOccurrences(of: ".m4a", with: "")
    
    let formatter = DateFormatter()
    formatter.dateFormat = "yyyy-MM-dd HH:mm"
    let timestamp = formatter.string(from: Date())
    let title = "Recording \(timestamp)"
    
    let newRecording = Recording(
        id: id,
        title: title,
        duration: duration,
        audioFilePath: filename,
        categoryTag: "#NOTE"
    )
```

- ✅ Entry appears in RecordingListView via @Query(sort: \Recording.createdAt, order: .reverse) - auto-refreshes
